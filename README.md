import random

print("Привіт! Це твоя перша гра з вгадування чисел.\nВ тебе є 7 спроб, щоб вгадати число. Давай почнемо!")

low = int(input("Ввведіть число від якого будемо вибирати: "))
high = int(input("Введіть число до якого будемо вибирати: "))

print(f"\nУ вас є 7 шансів, щоб вгадати число між числами {low} і {high}. Давай почнемо!")

num = random.randint(low, high) 
ch = 7                        # Кількість дозволених шансів на вгадування
gc = 0                        # Лічильник вгадувань

while gc < ch:
    gc += 1
    guess = int(input('Використай шанс і введи своє число: '))

    if guess == num:
        print(f'Правильно! Число є {num}. Ти вгадав це число з {gc} спроби.')
        break

    elif gc >= ch and guess != num:
        print(f'Як шкода!Число було {num}. Наступний раз пощастить більше.')

    elif guess > num:
        print('Занадто велике число! Спробуй нижче число.')

    elif guess < num:
        print('Занадто низьке число! Спробуй вище число.')
