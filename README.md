import random

print("Привіт! Це твоя перша гра з вгадування чисел.\nВ тебе є 7 спроб, щоб вгадати число. Давай почнемо!")

нижче_число = int(input("Ввведіть число від якого будемо вибирати: "))
вище_число = int(input("Введіть число до якого будемо вибирати: "))

print(f"\nУ вас є 7 шансів, щоб вгадати число між числами {нижче_число} і {вище_число}. Давай почнемо!")

загадане_число = random.randint(нижче_число, вище_число) 
кількість_шансів = 7
лічильник_вгадувань = 0

while лічильник_вгадувань < кількість_шансів:
    лічильник_вгадувань += 1
    вгадане_число = int(input('Використай шанс і введи своє число: '))

    if вгадане_число == загадане_число:
        print(f'Правильно! Число є {загадане_число}. Ти вгадав це число з {лічильник_вгадувань} спроби.')
        break

    elif лічильник_вгадувань >= кількість_шансів and вгадане_число != загадане_число:
        print(f'Як шкода!Число було {загадане_число}. Наступний раз пощастить більше.')

    elif вгадане_число > загадане_число:
        print('Занадто велике число! Спробуй нижче число.')

    elif вгадане_число < загадане_число:
        print('Занадто низьке число! Спробуй вище число.')
